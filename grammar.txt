SINGLE_DECLARE -> id col TYPE
MULTI_DECLARE -> list of variables VAR_LIST col TYPE semicol
TYPE -> PRIM_TYPE semicol
TYPE -> ARRAY
ARRAY -> RECTANGULAR
ARRAY -> JAGGED
PRIM_TYPE -> integer
PRIM_TYPE -> boolean
PRIM_TYPE -> real
RECTANGULAR -> array ARRAY_RANGE of integer
JAGGED -> jagged array sq_op num ddot num sq_cl JAGGED_INIT
JAGGED_INIT -> sq_op sq_cl JAGGED_INIT_DASH
JAGGED_INIT_DASH -> TD_INIT
JAGGED_INIT_DASH -> THD_INIT
TD_INIT -> of integer semicol TD_VALS
THD_INIT -> sq_op sq_cl of integer semicol THD_VALS
TD_VALS -> r1 sp_op num sq_cl col size num col values cb_op INT_LIST cb_cl TD_VALS_DASH
TD_VALS_DASH -> TD_VALS
TD_VALS_DASH -> EPS
INT_LIST -> num INT_LIST_DASH
INT_LIST_DASH -> EPS
INT_LIST_DASH -> semicol INT_LIST
THD_VALS -> r1 sq_op num sq_cl col size num col values cb_op INT_LIST_LIST cb_cl THD_VALS_DASH
THD_VALS_DASH -> THD_VALS
THD_VALS_DASH -> EPS
INT_LIST_LIST -> INT_VAR_LIST INT_LIST_LIST_DASH
INT_LIST_LIST_DASH -> semicol INT_LIST_LIST
INT_LIST_LIST_DASH -> EPS
INT_VAR_LIST -> num INT_VAR_LIST_DASH
INT_VAR_LIST_DASH -> EPS
INT_VAR_LIST_DASH -> INT_VAR_LIST
ARRAY_RANGE -> sq_op IDX ddot IDX sq_cl ARRAY_RANGE_DASH
ARRAY_RANGE_DASH -> ARRAY_RANGE
ARRAY_RANGE_DASH -> EPS
IDX -> id
IDX -> num
VAR_LIST -> id MULT_ID
MULT_ID -> id MULT_ID_DASH
MULT_ID_DASH -> EPS
MULT_ID_DASH -> MULT_ID

EXPR -> TERM EXPR_DASH

EXPR_DASH -> EPS

ASO_OP -> sub_op
ASO_OP -> or_op
