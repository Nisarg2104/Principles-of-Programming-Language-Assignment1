PROGRAM -> program rb_op rb_cl cb_op STATEMENTS cb_cl
STATEMENTS -> STATEMENT STATEMENTS
STATEMENTS -> STATEMENT
STATEMENT -> DECLARE_STATEMENT
STATEMENT -> ASSIGN_STATEMENT
DECLARE_STATEMENT -> SINGLE_DECLARE
DECLARE_STATEMENT -> MULTI_DECLARE
SINGLE_DECLARE -> declare id col TYPE
MULTI_DECLARE -> declare list of variables VAR_LIST col TYPE semicol
TYPE -> PRIM_TYPE semicol
TYPE -> ARRAY
ARRAY -> RECTANGULAR
ARRAY -> JAGGED
PRIM_TYPE -> integer
PRIM_TYPE -> boolean
PRIM_TYPE -> real
RECTANGULAR -> array ARRAY_RANGE of integer
JAGGED -> jagged array sq_op num ddot num sq_cl JAGGED_INIT
JAGGED_INIT -> 2D_INIT
JAGGED_INIT -> 3D_INIT

2D_INIT -> sq_op sq_cl of integer semicol 2D_VALS
3D_INIT -> sq_op sq_cl sq_op sq_cl of integer semicol 3D_VALS

2D_VALS -> R1 sp_op num sq_cl col size num col values cb_op INT_LIST cb_cl 2D_VALS
2D_VALS -> R1 sq_op num sq_cl col size num col values cb_op INT_LIST cb_cl
INT_LIST -> num semicol INT_LIST
INT_LIST -> num

3D_VALS -> R1 sq_op num sq_cl col size num col values cb_op INT_LIST_LIST cb_cl 3D_VALS
3D_VALS -> R1 sq_op num sq_cl col size num col values cb_op INT_LIST_LIST cb_cl
INT_LIST_LIST -> VAR_LIST semicol INT_LIST_LIST
INT_LIST_LIST -> VAR_LIST
VAR_LIST -> num VAR_LIST
VAR_LIST -> num


ARRAY_RANGE -> sq_op IDX ddot IDX sq_cl ARRAY_RANGE
ARRAY_RANGE -> sq_op IDX ddot IDX sq_cl

IDX -> id
IDX -> num
VAR_LIST -> id MULT_ID
MULT_ID -> id MULT_ID
MULT_ID -> id


ASSIGN_STATEMENT -> id EQ_OP EXPR
EXPR -> EXPR AS_OP TERM
EXPR -> TERM
AS_OP -> add_op
AS_OP -> sub_op
TERM -> TERM MD_OP IDX
TERM -> IDX		//typechecking at this stage or run time?
MD_OP -> mul_op
MD_OP -> div_op
